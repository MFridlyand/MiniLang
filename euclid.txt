function fib ( n )
{
  if n == 0
  {
	return 0
  }
  if n == 1
  {
   return 1
  }
  return call fib ( n - 1 ) + call fib ( n - 2 )
}

function gcd ( m , n )
{
	while  m != n 
	{
		if m > n 
			{ m = m - n }
		else
			{ n = n - m }
	}
	return m
}
