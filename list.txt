function reverse_list ( root )
{
	var newRoot = 0
	var nextNode = 0
	while root != 0
	{
		nextNode = call array_get ( root , 1 )
		call array_set ( root , 1 , newRoot )
		newRoot = root
		root = nextNode
	}
	return newRoot
}
